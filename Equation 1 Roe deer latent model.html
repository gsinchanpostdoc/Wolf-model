def load_vegetation_for_year(year, grid_ids, veg_list):
    """
    Loads all vegetation variables for a specific year and merges them 
    into a single dataframe aligned with the provided GridIDs.
    """
    # Start with a DataFrame of just the GridIDs we need
    veg_df = pd.DataFrame({'GridID': grid_ids})
    
    for v_type in veg_list:
        # Construct path: .../upscaled/VegName/VegName_YYYY.csv
        v_path = os.path.join(veg_base_dir, v_type, f"{v_type}_{year}.csv")
        
        if os.path.exists(v_path):
            try:
                # Read veg file
                df_v = pd.read_csv(v_path)
                
                # Keep only GridID and fraction column
                # Assuming column is named 'landCoverFrac' based on previous steps
                if 'landCoverFrac' in df_v.columns:
                    df_v = df_v[['GridID', 'landCoverFrac']]
                else:
                    # Fallback if column name differs (e.g. just 'frac')
                    # Take the last column or specific one if known
                    df_v = df_v[['GridID', df_v.columns[-1]]]
                
                # Rename fraction column to veg type name
                df_v.rename(columns={'landCoverFrac': v_type}, inplace=True)
                
                # Merge into main veg_df
                veg_df = pd.merge(veg_df, df_v, on='GridID', how='left')
                
            except Exception as e:
                print(f"    Warning: Could not read {v_type} for {year}: {e}")
                veg_df[v_type] = 0.0 # Fill missing file with 0
        else:
            # If file missing (e.g. year out of range), fill with 0
            veg_df[v_type] = 0.0
            
    # Fill any NaNs (if grid missing in veg file) with 0
    veg_df.fillna(0, inplace=True)
    return veg_df

def bayesian_logit_model(X_train, y_train, X_pred, veg_types):
    """
    Implements Bayesian framework for the logit model:
    Logit(D(g,t)^s) = _0^s + ^s_ * V^(g,t)
    
    Where:
    - D(g,t)^s is density of species s in grid g at time t
    - V^(g,t) is vegetation cover fraction
    - _0 and  are coefficients
    """
    
    # Transform density to logit space (handling zeros)
    epsilon = 1e-6  # Small value to avoid log(0)
    y_train_clipped = np.clip(y_train, epsilon, 1 - epsilon)
    y_logit = np.log(y_train_clipped / (1 - y_train_clipped))
    
    # Calculate Bayesian priors (using empirical Bayes approach)
    # Prior for _0: mean of logit-transformed densities
    beta_0_prior = np.mean(y_logit)
    beta_0_var = np.var(y_logit)
    
    # Prior for  coefficients: assuming weak informativeness
    n_features = X_train.shape[1]
    beta_prior = np.zeros(n_features)
    beta_var = np.ones(n_features) * 0.5
    
    # Posterior estimation using weighted least squares approximation
    # Weight observations by their information content
    weights = y_train_clipped * (1 - y_train_clipped)  # Variance function for logit
    
    # Add intercept column
    X_train_intercept = np.column_stack([np.ones(len(X_train)), X_train])
    
    # Weighted regression for posterior mean
    W = np.diag(weights)
    
    # Posterior covariance
    prior_precision = np.diag(1 / np.append(beta_0_var, beta_var))
    data_precision = X_train_intercept.T @ W @ X_train_intercept
    posterior_precision = prior_precision + data_precision
    posterior_cov = np.linalg.inv(posterior_precision)
    
    # Posterior mean
    prior_mean = np.append(beta_0_prior, beta_prior)
    posterior_mean = posterior_cov @ (prior_precision @ prior_mean + 
                                      X_train_intercept.T @ W @ y_logit)
    
    # Extract coefficients
    beta_0_post = posterior_mean[0]
    beta_post = posterior_mean[1:]
    
    # Generate predictions with uncertainty
    X_pred_intercept = np.column_stack([np.ones(len(X_pred)), X_pred])
    
    # Point predictions in logit space
    logit_pred = X_pred_intercept @ posterior_mean
    
    # Prediction uncertainty (from posterior distribution)
    pred_var = np.diag(X_pred_intercept @ posterior_cov @ X_pred_intercept.T)
    pred_std = np.sqrt(pred_var)
    
    # Transform back from logit to density space
    density_pred = 1 / (1 + np.exp(-logit_pred))
    
    # Calculate credible intervals (95%)
    logit_lower = logit_pred - 1.96 * pred_std
    logit_upper = logit_pred + 1.96 * pred_std
    density_lower = 1 / (1 + np.exp(-logit_lower))
    density_upper = 1 / (1 + np.exp(-logit_upper))
    
    # Return results
    results = {
        'predictions': density_pred,
        'lower_CI': density_lower,
        'upper_CI': density_upper,
        'beta_0': beta_0_post,
        'beta': dict(zip(veg_types, beta_post)),
        'posterior_cov': posterior_cov
    }
    
    return results

def comprehensive_model_comparison(train_df, veg_types, density_col, output_dir):
    """
    Compare Bayesian Logit vs Random Forest models across multiple metrics
    """
    
    print("\n" + "="*60)
    print("COMPREHENSIVE MODEL COMPARISON")
    print("="*60)
    
    # Prepare data
    X = train_df[veg_types].values
    y = train_df[density_col].values
    
    # Initialize results storage
    comparison_results = {
        'Bayesian_Logit': {},
        'Random_Forest': {}
    }
    
    # -------------------------------------------------------------------------
    # 1. CROSS-VALIDATION PERFORMANCE
    # -------------------------------------------------------------------------
    print("\n1. 5-Fold Cross-Validation Performance:")
    print("-" * 40)
    
    kf = KFold(n_splits=5, shuffle=True, random_state=42)
    
    bayesian_scores = {'r2': [], 'rmse': [], 'mae': [], 'mape': []}
    rf_scores = {'r2': [], 'rmse': [], 'mae': [], 'mape': []}
    
    for train_idx, test_idx in kf.split(X):
        X_train_cv, X_test_cv = X[train_idx], X[test_idx]
        y_train_cv, y_test_cv = y[train_idx], y[test_idx]
        
        # Bayesian Logit predictions
        bay_results = bayesian_logit_model(X_train_cv, y_train_cv, X_test_cv, veg_types)
        y_pred_bay = bay_results['predictions']
        
        # Random Forest predictions
        rf_model = RandomForestRegressor(n_estimators=100, random_state=42, n_jobs=-1)
        rf_model.fit(X_train_cv, y_train_cv)
        y_pred_rf = rf_model.predict(X_test_cv)
        
        # Calculate metrics for Bayesian
        bayesian_scores['r2'].append(r2_score(y_test_cv, y_pred_bay))
        bayesian_scores['rmse'].append(np.sqrt(mean_squared_error(y_test_cv, y_pred_bay)))
        bayesian_scores['mae'].append(mean_absolute_error(y_test_cv, y_pred_bay))
        mape_bay = np.mean(np.abs((y_test_cv - y_pred_bay) / (y_test_cv + 1e-6))) * 100
        bayesian_scores['mape'].append(mape_bay)
        
        # Calculate metrics for RF
        rf_scores['r2'].append(r2_score(y_test_cv, y_pred_rf))
        rf_scores['rmse'].append(np.sqrt(mean_squared_error(y_test_cv, y_pred_rf)))
        rf_scores['mae'].append(mean_absolute_error(y_test_cv, y_pred_rf))
        mape_rf = np.mean(np.abs((y_test_cv - y_pred_rf) / (y_test_cv + 1e-6))) * 100
        rf_scores['mape'].append(mape_rf)
    
    # Print CV results
    print("\nBayesian Logit Model:")
    print(f"  R Score:     {np.mean(bayesian_scores['r2']):.3f}  {np.std(bayesian_scores['r2']):.3f}")
    print(f"  RMSE:         {np.mean(bayesian_scores['rmse']):.3f}  {np.std(bayesian_scores['rmse']):.3f}")
    print(f"  MAE:          {np.mean(bayesian_scores['mae']):.3f}  {np.std(bayesian_scores['mae']):.3f}")
    print(f"  MAPE:         {np.mean(bayesian_scores['mape']):.1f}%  {np.std(bayesian_scores['mape']):.1f}%")
    
    print("\nRandom Forest Model:")
    print(f"  R Score:     {np.mean(rf_scores['r2']):.3f}  {np.std(rf_scores['r2']):.3f}")
    print(f"  RMSE:         {np.mean(rf_scores['rmse']):.3f}  {np.std(rf_scores['rmse']):.3f}")
    print(f"  MAE:          {np.mean(rf_scores['mae']):.3f}  {np.std(rf_scores['mae']):.3f}")
    print(f"  MAPE:         {np.mean(rf_scores['mape']):.1f}%  {np.std(rf_scores['mape']):.1f}%")
    
    # Store results
    comparison_results['Bayesian_Logit']['cv_metrics'] = bayesian_scores
    comparison_results['Random_Forest']['cv_metrics'] = rf_scores
    
    # -------------------------------------------------------------------------
    # 2. PREDICTION BEHAVIOR ANALYSIS
    # -------------------------------------------------------------------------
    print("\n2. Prediction Behavior Analysis:")
    print("-" * 40)
    
    # Train on full data for behavior analysis
    bay_full = bayesian_logit_model(X, y, X, veg_types)
    rf_full = RandomForestRegressor(n_estimators=100, random_state=42, n_jobs=-1)
    rf_full.fit(X, y)
    
    y_pred_bay_full = bay_full['predictions']
    y_pred_rf_full = rf_full.predict(X)
    
    # Analyze prediction ranges
    print("\nPrediction Range Analysis:")
    print(f"  Actual density range:        [{y.min():.3f}, {y.max():.3f}]")
    print(f"  Bayesian predictions range:  [{y_pred_bay_full.min():.3f}, {y_pred_bay_full.max():.3f}]")
    print(f"  RF predictions range:        [{y_pred_rf_full.min():.3f}, {y_pred_rf_full.max():.3f}]")
    
    # Check for unrealistic predictions
    bay_negative = np.sum(y_pred_bay_full < 0)
    rf_negative = np.sum(y_pred_rf_full < 0)
    bay_extreme = np.sum(y_pred_bay_full > y.max() * 1.5)
    rf_extreme = np.sum(y_pred_rf_full > y.max() * 1.5)
    
    print(f"\nUnrealistic Predictions:")
    print(f"  Bayesian - Negative values: {bay_negative}, Extreme values (>1.5x max): {bay_extreme}")
    print(f"  RF - Negative values: {rf_negative}, Extreme values (>1.5x max): {rf_extreme}")

    print(f"  Bayesian CI coverage: Provided by credible intervals")
    
    # RF uncertainty through prediction variance across trees
    rf_predictions = np.array([tree.predict(X) for tree in rf_full.estimators_])
    rf_std = np.std(rf_predictions, axis=0)

    rf_rmse_mean = np.mean(rf_scores['rmse'])
